 15    """Deep Potential embedding
 16
 17    FID : DEEPPOT
 18
 19    Reference
 20    ----------
 21    Zhang, L., Han, J., Wang, H., Car, R., & E, W. (2018). Deep Potential Molecular dynamics: A scalable model with the accuracy of quantum mechanics. Phys. Rev. Lett., 120(14), 143001. https://doi.org/10.1103/PhysRevLett.120.143001
 22
 23    Parameters
 24    ----------
 25    dim : int, default=64
 26        The dimension of the embedding.
 27    subdim : int, default=8
 28        The first dimensions to select for the embedding tensor product.
 29    radial_dim : Optional[int], default=None
 30        The dimension of the radial embedding for tensor combination. 
 31        If None, we use a neural net to combine chemical and radial information, like in the original DeepPot.
 32    embedding_key : str, default="embedding"
 33        The key to use for the output embedding in the returned dictionary.
 34    graph_key : str, default="graph"
 35        The key in the input dictionary that corresponds to the radial graph.
 36    species_encoding : dict, default={}
 37        The species encoding parameters.
 38    radial_basis : Optional[dict], default=None
 39        The radial basis parameters. If None, the radial basis is the s_ij like in the original DeepPot.
 40    embedding_hidden : Sequence[int], default=[64, 64, 64]
 41        The hidden layers of the embedding network.
 42    activation : Union[Callable, str], default=nn.silu
 43        The activation function.
 44    concatenate_species : bool, default=False
 45        Whether to concatenate the species encoding with the embedding.
 46    divide_distances : bool, default=True
 47        Whether to divide the switch by the distance in s_ij.
 48    species_order : Optional[Sequence[str]], default=None
 49        Species considered by the network when using species-specialized embedding network.
 50
 51    """
 52    _graphs_properties: Dict
 53    dim: int = 64
 54    subdim: int = 8
 55    radial_dim: Optional[int] = None
 56    embedding_key: str = "embedding"
 57    graph_key: str = "graph"
 58    species_encoding: dict = dataclasses.field(default_factory=dict)
 59    radial_basis: Optional[dict] = None
 60    embedding_hidden: Sequence[int] = dataclasses.field(
 61        default_factory=lambda: [64, 64, 64]
 62    )
 63    activation: Union[Callable, str] = nn.silu
 64    concatenate_species: bool = False
 65    divide_distances: bool = True
 66    species_order: Optional[Union[str,Sequence[str]]] = None
 67
 68    FID: str = "DEEPPOT"
 69
 70    @nn.compact
 71    def __call__(self, inputs):
 72        species = inputs["species"]
 73
 74        # species encoding
 75        if self.species_order is None or self.concatenate_species:
 76            onehot = SpeciesEncoding(**self.species_encoding, name="SpeciesEncoding")(
 77                species
 78            )
 79
 80        # Radial graph
 81        graph = inputs[self.graph_key]
 82        edge_src, edge_dst = graph["edge_src"], graph["edge_dst"]
 83        distances = graph["distances"][:, None]
 84        switch = graph["switch"][:, None]
 85        vec = graph["vec"] / distances
 86        sij = switch / distances if self.divide_distances else switch
 87        Rij = jnp.concatenate((sij, sij * vec), axis=-1)
 88
 89        # Radial BASIS
 90        cutoff = self._graphs_properties[self.graph_key]["cutoff"]
 91        if self.radial_basis is not None:
 92            radial_terms = RadialBasis(
 93                **{
 94                    **self.radial_basis,
 95                    "end": cutoff,
 96                    "name": f"RadialBasis",
 97                }
 98            )(graph["distances"])
 99        else:
100            radial_terms = sij
101
102        if self.species_order is not None:
103            Gij = ChemicalNet(
104                self.species_order,
105                [*self.embedding_hidden, self.dim],
106                activation=self.activation,
107            )((species[edge_dst], radial_terms))
108        elif self.radial_dim is not None:
109            Gij = FullyConnectedNet(
110                [*self.embedding_hidden, self.radial_dim], activation=self.activation
111            )(radial_terms)
112            Wa = self.param(
113                f"Wa",
114                nn.initializers.normal(
115                    stddev=1.0 / (Gij.shape[1] * onehot.shape[1]) ** 0.5
116                ),
117                (onehot.shape[1], Gij.shape[1], self.dim),
118            )
119            Gij = jnp.einsum(
120                "...i,...j,ijk->...k",
121                onehot[edge_dst],
122                Gij,
123                Wa,
124            )
125        else:
126            Gij = FullyConnectedNet(
127                [*self.embedding_hidden, self.dim], activation=self.activation
128            )(jnp.concatenate((radial_terms, onehot[edge_dst]), axis=-1))
129
130        GRi = jax.ops.segment_sum(
131            Gij[:, None, :] * Rij[:, :, None], edge_src, species.shape[0]
132        )
133        if self.subdim > 0:
134            GRisub = GRi[:, :, : self.subdim]
135
136            embedding = (
137                (GRi[:, :, :, None] * GRisub[:, :, None, :])
138                .sum(axis=1)
139                .reshape((species.shape[0], -1))
140            )
141        else:
142            GRisub = nn.Dense(self.dim, use_bias=False, name="Gri_linear")(GRi)
143            embedding = (GRi * GRisub).sum(axis=1)
144
145        if self.concatenate_species:
146            embedding = jnp.concatenate((onehot, embedding), axis=-1)
147
148        if self.embedding_key is None:
149            return embedding
150        return {**inputs, self.embedding_key: embedding}

Wraps parent module references in weak refs.

This prevents reference cycles from forming via parent links which can lead to accidental OOMs in eager mode due to slow garbage collection as well as spurious tracer leaks during jit compilation.

Note: "descriptors" are the underlying python mechanism for implementing dynamic @property decorators. We need to use a raw descriptor instead of the more common decorator in order to force that the appropriate getter/setter logic applies in subclasses even after various dataclass transforms.
